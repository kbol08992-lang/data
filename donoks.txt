    <title>Tampilkan Konten TXT</title>

<body>
    <h2>Isi File dones.txt</h2>
    <pre id="konten"></pre>

    <script>
        const url = "https://raw.githubusercontent.com/kbol08992-lang/data/refs/heads/main/dones.txt";

        fetch(url)
            .then(response => response.text())
            .then(data => {
                document.getElementById("konten").textContent = data;
            })
            .catch(error => {
                document.getElementById("konten").textContent = "Gagal memuat data: " + error;
            });
    </script>
</body>
